def fibo:int( n, a:int )
{
	if( n == 0 || n == 1)
		return 1;
	return fibo(n-2) + fibo(n-1);
}

def expBinaria:long( base, exp: long )
{
        let respuesta:long = 1;
        while(exp)
        {
            if(exp & 1)
                respuesta *= base;
            base *= base;
            exp>>1;
	}
	return base;
}
